  const glyph = ICONS[name];
  if (!glyph) return null;
  return (
    <svg
      className={`ico ${className}`.trim()}
      width={size}
      height={size}
      viewBox={VB}
      fill="none"
      stroke="currentColor"

      {...rest}
    >
      {title && <title>{title}</title>}
      {glyph}
    </svg>
  );
}
